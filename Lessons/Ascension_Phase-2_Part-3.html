import openai
def find_vulnerabilities(domain):
    prompt = f"Analyze the domain {domain} for common vulnerabilities."
    response = openai.Completion.create(
        engine="text-davinci-003",
        prompt=prompt,
        max_tokens=500
    )
    print(response.choices[0].text)

find_vulnerabilities("example.com")
